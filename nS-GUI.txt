local Library = loadstring(game:HttpGet("https://raw.githubusercontent.com/xHeptc/Kavo-UI-Library/main/source.lua"))()

local Window = Library.CreateLib("nS-GUI", "BloodTheme")

-- Mains

local Painel = Window:NewTab("Painel")

local Scripts = Painel:NewSection("Scripts")

local AutoClick = Window:NewTab("AutoClick")

local AutoClick = AutoClick:NewSection("AutoClick")

-- Executes:

Scripts:NewButton("Blox Fruits (HOHO)", "Execute HoHo Hub", function()

    loadstring(game:HttpGet('https://raw.githubusercontent.com/acsu123/HOHO_H/main/Loading_UI'))()

end)

Scripts:NewButton("Shadovis RPG (NOVICE)", "Execute Novice Hub", function()

    loadstring(game:HttpGet("https://raw.githubusercontent.com/Amphvptere/Novice/main/SHADrpg"))()

end)

AutoClick:NewToggle("AutoClick", "", function(state)

    if state then

        while true do

        local Event = game:GetService("ReplicatedStorage"). Events.Click

        Event:FireServer()

        wait()

        end

        end)

        print("AutoClick On")

    else

        print("AutoClick Off")

    end

end)
